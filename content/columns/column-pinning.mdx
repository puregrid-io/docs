You can pin any column with the `pinned: true` option, there is no stipulation that the column has to be on the far right or left. When it touches either side of the grid edge it will stick there.

```ts:title=Pinning+a+column
const { columns, setColumns } = useColumns<ImportantData>([
  {
    key: 'id',
    header: 'ID',
    width: 60,
    getValue: d => d.id,
    pinned: true,
  },
  {
    key: 'veryWideThing',
    header: 'Very Wide Thing',
    width: 1200,
    getValue: d => d.veryWideThing,
  },
  {
    key: 'killSwitch',
    header: 'Kill it',
    width: 64,
    getValue: d => d.killSwitch,
    align: ColumnAlign.End,
    cellComponent: 'checkbox',
    pinned: true,
  },
  {
    key: 'anotherVeryWideThing',
    header: 'Another Very Wide Thing',
    width: 1200,
    getValue: d => d.veryWideThing,
  },
]);
```
